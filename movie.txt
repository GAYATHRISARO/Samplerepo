Movie Basicquery:
-----------------

1)select mov_title,mov_year 
  from movie;
2)select mov_year 
  from movie 
  where mov_title=American Beauty;
3)select mov_title 
  from movie 
  where mov_year=1999;
4)select mov_title 
  from movie 
  where mov_year < 1998;
5)select rev_name 
  from reviewer union
  (select movie mov_title from movie);
6)select r.rev_name 
  from reviewer r,rating ra 
  where r.rev_id=ra.rev_id and ra.rev_stars >= 7;
7)select mov_title
  from movie
  where mov_id not in(select mov_id from rating);
8)select mov_title
  from movie
  where mov_id in(905,907,917);
9)select mov_title,mov_year
  from movie
  where mov_title="Boogie Nights";
10)select act_id 
   from actor
   where act_fname like 'Woody%' and act_lname like 'Allen%';
